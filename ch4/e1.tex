\subsection{4.1-2}
    \begin{codebox}
        \Procname{$\proc{Maxmium-Subarray}(A)$}
        \li $t \gets -\infty$
        \li \For $i \gets 1 \To \attrib{A}{length}-1$
            \Do
        \li     \For $j \gets 2 \To \attrib{A}{length}$
                \Do
        \li         $t \gets \max(j, A[j] - A[i])$
                \End
        \li \Return t
            \End
    \end{codebox}
